Write `frontend/src/components/canvas/PointCloud.tsx` — THREE.Points with custom ShaderMaterial.

CRITICAL: Use THREE.Points NOT InstancedMesh (per RESEARCH.md). The UI-SPEC mentions InstancedMesh but research overrides this for performance.

The vertex shader reads per-point `aSize`, `aOpacity`, `aColor` attributes:
```glsl
attribute float aSize;
attribute float aOpacity;
attribute vec3 aColor;
varying float vOpacity;
varying vec3 vColor;
uniform float uSizeMultiplier;

void main() {
  vOpacity = aOpacity;
  vColor = aColor;
  vec4 mvPosition = modelViewMatrix * vec4(position, 1.0);
  gl_PointSize = aSize * uSizeMultiplier * (300.0 / -mvPosition.z);
  gl_Position = projectionMatrix * mvPosition;
}
```

Fragment shader clips to circle via gl_PointCoord:
```glsl
varying float vOpacity;
varying vec3 vColor;
void main() {
  float dist = length(gl_PointCoord - vec2(0.5));
  if (dist > 0.5) discard;
  float alpha = smoothstep(0.5, 0.35, dist) * vOpacity;
  gl_FragColor = vec4(vColor, alpha);
}
```

interface PointCloudProps {

In the component body:

Float32Array builder helper (inline in App.tsx):
```typescript
function buildBuffers(points: ScatterPoint[], genreColors: Record<string, string>) {
  const n = points.length
  const positions = new Float32Array(n * 3)
  const colors = new Float32Array(n * 3)
  const sizes = new Float32Array(n)
  const opacities = new Float32Array(n)
  for (let i = 0; i < n; i++) {
    const p = points[i]
    positions[i * 3] = p.x
    positions[i * 3 + 1] = p.y
    positions[i * 3 + 2] = p.z
    const hex = genreColors[p.genre] ?? '#888888'
    const color = new THREE.Color(hex)
    colors[i * 3] = color.r; colors[i * 3 + 1] = color.g; colors[i * 3 + 2] = color.b
    sizes[i] = 2.0 + p.tfidf_weight * 8.0   // 2–10 units; ShaderMaterial scales by depth
    opacities[i] = Math.max(0.08, p.tfidf_weight) // dim but never invisible
  }
  return { positions, colors, sizes, opacities }
}
```
