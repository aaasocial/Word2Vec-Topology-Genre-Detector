**Step A — Create `frontend/src/hooks/useCorpusBooks.ts`**

```tsx
import { useQuery } from '@tanstack/react-query'
import { api } from '@/lib/api'   // adjust to actual project's fetch helper

export type CorpusBookFull = {
  gutenberg_id: string
  title: string
  author: string
  genre: string
  word_count: number
  color: string
  top_10_tfidf_words: string[]
}

/**
 * Fetch all books in a genre with full metadata for BookSlider.
 * Per CONTEXT.md D-13: staleTime Infinity since corpus only changes on retrain.
 */
export function useCorpusBooks(genre: string | null) {
  return useQuery<CorpusBookFull[]>({
    queryKey: ['corpus', 'genres', genre, 'books'],
    queryFn: async () => {
      if (!genre) return []
      const res = await fetch(`/api/corpus/genres/${encodeURIComponent(genre)}/books`)
      if (!res.ok) {
        if (res.status === 404) return []   // unknown genre = empty list
        throw new Error(`useCorpusBooks(${genre}): ${res.status}`)
      }
      return res.json()
    },
    enabled: !!genre,
    staleTime: Infinity,
  })
}
```

| Boundary | Description |

| Untrusted HTTP client → FastAPI route | `/api/corpus/genres/{genre}/books` accepts `{genre}` as a path parameter. This IS the primary new attack surface in Phase 6. |
| Backend file system → in-process cache | `corpus/books.yaml` and `data/corpus_metadata.json` are loaded once at module import. If the deploy environment lets an attacker substitute these files, they control the response — but this is the same threat as any deploy-time supply-chain compromise. Out of scope for endpoint hardening. |
| Frontend → backend | `useCorpusBooks` URL-encodes the genre. No SQL, no shell, no eval. |
