**Step A — Modify `backend/cache/store.py`**

Replace the existing `cache_key` function with the new keyword-only signature. Add helper functions:

```python
import hashlib
import json
from pathlib import Path
from typing import Any, Optional, Union

import numpy as np
import yaml

CACHE_DIR = Path(__file__).resolve().parents[2] / 'data' / 'cache'
CORPUS_PATH = Path(__file__).resolve().parents[2] / 'corpus' / 'books.yaml'
MODELS_DIR = Path(__file__).resolve().parents[2] / 'data' / 'models'


def cache_key(
    step_name: str,
    params: dict,
    *,
    corpus_hash: str,
    w2v_model_sha256: str,
) -> str:
    """Generate deterministic cache key from step name, params, corpus identity, and model identity.

    BUG-05 / PITFALLS.md §1: corpus_hash and w2v_model_sha256 are REQUIRED to prevent
    serving cached artifacts against a rotated Word2Vec embedding space.
    """
    canonical = json.dumps(
        {
            'step': step_name,
            'params': params,
            'corpus_hash': corpus_hash,
            'w2v_model_sha256': w2v_model_sha256,
        },
        sort_keys=True,
        default=str,
    )
    return hashlib.sha256(canonical.encode()).hexdigest()


def compute_corpus_hash(corpus_path: Path = CORPUS_PATH) -> str:
    """SHA-256 of corpus/books.yaml bytes. Hashes the manifest, not derived caches."""
    if not corpus_path.exists():
        raise FileNotFoundError(f"Corpus manifest not found: {corpus_path}")
    return hashlib.sha256(corpus_path.read_bytes()).hexdigest()


def compute_w2v_model_sha256(window: int, models_dir: Path = MODELS_DIR) -> str:
    """SHA-256 of the Word2Vec model file for the given window size.

    Filename convention discovered from data/models/. Adjust pattern below
    to whatever name the repo actually uses — verify with `ls data/models/`.
    """
    # Common filename patterns to try in order. Adjust to whatever exists.
    candidates = [
        models_dir / f'word2vec_window{window}.model',
        models_dir / f'w2v_window{window}.model',
        models_dir / f'word2vec_w{window}.model',
    ]
    for path in candidates:
        if path.exists():
            return hashlib.sha256(path.read_bytes()).hexdigest()
    raise FileNotFoundError(
        f"Word2Vec model not found for window={window}. Tried: {candidates}"
    )
```

"""SVM lineage save/load with hash-match guard.

Defense-in-depth on top of BUG-05 cache_key fix (D-25). PITFALLS.md §1:
'Pin the SVM training data lineage. Save alongside the SVM: model hash,
corpus manifest hash, feature-track normalization stats, α. Refuse to
load an SVM whose lineage doesn't match the currently-loaded W2V model.'
"""
import json
from dataclasses import dataclass, asdict
from pathlib import Path
from typing import Any

import joblib


class LineageMissingError(RuntimeError):
    """Raised when an SVM file exists but its .lineage.json sidecar does not."""


class LineageMismatchError(RuntimeError):
    """Raised when a loaded SVM's saved lineage does not match the current pipeline state."""


@dataclass(frozen=True)
class SvmLineage:
    model_hash: str           # sha256 of the W2V model file used at training time
    corpus_hash: str          # sha256 of corpus/books.yaml at training time
    alpha: float              # feature-track weighting at training time
    normalization_stats: dict  # per-track L2 norms or means (small dict, json-safe)


def _sidecar_path(svm_path: Path) -> Path:
    return svm_path.with_suffix(svm_path.suffix + '.lineage.json')


def save_svm_lineage(
    svm_path: Path,
    *,
    model_hash: str,
    corpus_hash: str,
    alpha: float,
    normalization_stats: dict,
) -> Path:
    lineage = SvmLineage(
        model_hash=model_hash,
        corpus_hash=corpus_hash,
        alpha=alpha,
        normalization_stats=normalization_stats,
    )
    sidecar = _sidecar_path(svm_path)
    sidecar.write_text(json.dumps(asdict(lineage), sort_keys=True, indent=2))
    return sidecar


def load_svm_with_lineage_check(
    svm_path: Path,
    *,
    current_model_hash: str,
    current_corpus_hash: str,
) -> Any:
    """Load the SVM only if its lineage matches the current pipeline state."""
    sidecar = _sidecar_path(svm_path)
    if not sidecar.exists():
        raise LineageMissingError(
            f"SVM lineage sidecar missing: {sidecar}. "
            f"Retrain via scripts/06_validate.py to write lineage."
        )
    saved = json.loads(sidecar.read_text())
    if saved['model_hash'] != current_model_hash:
        raise LineageMismatchError(
            f"SVM trained against model_hash={saved['model_hash'][:8]}... "
            f"but current model_hash={current_model_hash[:8]}.... "
            f"Retrain SVM (PITFALLS.md §1)."
        )
    if saved['corpus_hash'] != current_corpus_hash:
        raise LineageMismatchError(
            f"SVM trained against corpus_hash={saved['corpus_hash'][:8]}... "
            f"but current corpus_hash={current_corpus_hash[:8]}.... "
            f"Retrain SVM (PITFALLS.md §1)."
        )
    return joblib.load(svm_path)
```

import json
import joblib
from pathlib import Path
import pytest
import numpy as np
from sklearn.svm import SVC

from backend.cache.lineage import (
    save_svm_lineage,
    load_svm_with_lineage_check,
    LineageMissingError,
    LineageMismatchError,
)


@pytest.fixture
def trained_svm(tmp_path):
    X = np.random.randn(20, 5)
    y = np.array([0,1]*10)
    svm = SVC(probability=True).fit(X, y)
    path = tmp_path / 'svm.joblib'
    joblib.dump(svm, path)
    return path


def test_save_writes_sidecar(trained_svm):
    sidecar = save_svm_lineage(
        trained_svm,
        model_hash='M1', corpus_hash='C1', alpha=0.5,
        normalization_stats={'topology_l2_mean': 1.0, 'cluster_l2_mean': 1.0},
    )
    assert sidecar.exists()
    data = json.loads(sidecar.read_text())
    assert data['model_hash'] == 'M1'
    assert data['corpus_hash'] == 'C1'
    assert data['alpha'] == 0.5
    assert 'topology_l2_mean' in data['normalization_stats']


def test_load_with_matching_lineage_succeeds(trained_svm):
    save_svm_lineage(trained_svm, model_hash='M1', corpus_hash='C1', alpha=0.5, normalization_stats={})
    svm = load_svm_with_lineage_check(trained_svm, current_model_hash='M1', current_corpus_hash='C1')
    assert svm is not None


def test_load_with_mismatched_model_raises(trained_svm):
    save_svm_lineage(trained_svm, model_hash='M1', corpus_hash='C1', alpha=0.5, normalization_stats={})
    with pytest.raises(LineageMismatchError, match='model_hash'):
        load_svm_with_lineage_check(trained_svm, current_model_hash='M2', current_corpus_hash='C1')


def test_load_with_mismatched_corpus_raises(trained_svm):
    save_svm_lineage(trained_svm, model_hash='M1', corpus_hash='C1', alpha=0.5, normalization_stats={})
    with pytest.raises(LineageMismatchError, match='corpus_hash'):
        load_svm_with_lineage_check(trained_svm, current_model_hash='M1', current_corpus_hash='C2')


def test_load_with_missing_sidecar_raises(trained_svm):
    # No save_svm_lineage call — sidecar absent.
    with pytest.raises(LineageMissingError):
        load_svm_with_lineage_check(trained_svm, current_model_hash='M1', current_corpus_hash='C1')
```

**Anti-patterns to avoid:**
- Storing the full `normalization_stats` ndarrays (use small json-safe dicts of floats)
- Loading the SVM AND THEN checking lineage (call must refuse load on mismatch — joblib.load is only called after the check passes)
- Using a single exception class for "missing" vs "mismatched" (operators need to distinguish "regenerate lineage" from "retrain SVM")
- Computing `current_model_hash` lazily inside `load_svm_with_lineage_check` (the caller must pass it; this keeps the function pure and testable)

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Disk model artifacts (`data/models/`) → in-memory pipeline | The W2V model file and SVM file are loaded server-side. `compute_w2v_model_sha256` and `compute_corpus_hash` read from these paths. If an attacker could substitute either file, they control predictions. This is the same threat surface as any compromised deploy — mitigated at deploy-pipeline level, not at runtime. |
| File: `corpus/books.yaml` → cache_key inputs | Hashed verbatim. Tampering changes the hash, which forces a cache miss (correct behavior), not poisoning. |

The D-24 smoke test verifies "old cache + new model = miss" before Phase 8 starts. |
